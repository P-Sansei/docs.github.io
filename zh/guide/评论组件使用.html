# 添加 Valine.vue 组件

# Valine.vue

<template>
  <!-- Valine 无后端评论系统 组件 -->
  <div v-if="valine">
    <h3 class="valine">留言板:</h3>
    <!-- id 将作为查询条件 -->
    <!-- 禁用阅读量：原因是项目base路径以及部署github-pages 项目名不规范
         正确示例：base: '/', 项目名: documents.github.io
         有域名的可以配置下域名： 解析方式: CNAME 参考域名解析官网
     -->
    <!-- <span :id="path.path" class="leancloud-visitors" :data-flag-title="path.title">
      <em class="post-meta-item-text">阅读量 </em>
      <i class="leancloud-visitors-count">1000000+</i>
    </span> -->
    <div id="vcomments"></div>
  </div>
</template>

<script>
export default {
  computed: {
    valine: function() {
      let { valine } = this.$frontmatter;
      if (typeof valine !== "undefined") {
        return valine;
      }
      return true;
    },
    path: function() {
      let { path = "/documents/", title = "首页" } = this.$page;
      return { path, title };
    }
  },
  mounted() {
    this.renderValine();
  },
  watch: {
    $route: {
      handler: function(to, from) {
        if (to.path !== from.path) {
          this.$nextTick(() => {
            this.renderValine();
          });
        }
      }
    }
  },
  methods: {
    renderValine() {
      const Valine = require("valine");
      if (typeof window !== undefined) {
        this.window = window;
        window.AV = require("leancloud-storage");
      }
      new Valine({
        el: "#vcomments",
        appId: "wKVcbMAXgbG3rvQ7i0X7lOl1-gzGzoHsz",
        appKey: "9KIGUfwtK8h4gHSw3YG5FJ10",
        notify: false,
        verify: false,
        avatar: "retro",
        path: window.location.pathname,
        meta: ["nick", "mail", "link"],
        pageSize: 10,
        visitor: true, // 阅读量统计
        placeholder: "欢迎留言..."
      });
    }
  }
};
</script>

注意

vuepress 官方文档 中介绍在 .vueprss/components 中创建组件, 可以在 Markdown 页面中直接如下使用

<!-- Valine -->
<Valine></Valine>
1
2

这样文档需要的评论的组件地方都需要添加这行代码, 重复添加很是麻烦。 本着 事不过三,三则重构的原则,进行优化。

# 使用自定义主题( 嘿嘿~： 其实就是在默认主题上修改)

  • 在 .vuepress 文件夹下创建 theme 文件夹, 在 node_modules\@vuepress\theme-default 中拷贝所有文件到 theme 中

  • 在 global-components 中添加 Valine.vue, global-components 会自动注册成全局异步组件

  • 在 components/Page.vue 添加使用组件

......
<div class="page-nav" v-if="prev || next">
  <p class="inner">
    <span v-if="prev" class="prev">
      ←
      <router-link v-if="prev" class="prev" :to="prev.path">{{ prev.title || prev.path }}</router-link>
    </span>

    <span v-if="next" class="next">
      <router-link v-if="next" :to="next.path">{{ next.title || next.path }}</router-link>→
    </span>
  </p>
  <!-- Valine 评论组件 Begin -->
  <Valine></Valine>
  <!-- Valine 评论组件 End -->
</div>
......
